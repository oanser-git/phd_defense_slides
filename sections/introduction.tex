\begin{frame}{From signatures to ML-based NIDS}
	\small
	\begin{columns}[T,onlytextwidth]
		\column{0.48\textwidth}
		\begin{block}{Rule-based NIDS}
			\begin{itemize}
				\setlength\itemsep{0.6em}
				\item Tools: Snort, Suricata, Zeek
				\item Detect only known attack patterns or protocol violations, leaving unknown ones undetected.
				\item Encrypted traffic limits payload-based rules
			\end{itemize}
		\end{block}

		\column{0.48\textwidth}
		\begin{block}{ML-based NIDS}
			\begin{itemize}
				\setlength\itemsep{0.6em}
				\item Example: Random Forest (RF) on flows
				\item Learn traffic patterns from historical data to generalize beyond exact rules.
				\item Can learn complex traffic patterns even under encryption
			\end{itemize}
		\end{block}
	\end{columns}

	\vspace{0.25em}
	\begin{beamercolorbox}[wd=\linewidth,sep=0.9ex,rounded=true]{block body}
		\centering
		\textbf{Operational reality: mostly hybrid\sliderefmark{1}.} E.g. SOCs use \textbf{Suricata/Zeek} for known alerts + \textbf{ML on flows} to flag suspicious HTTPS C\&C behavior.
	\end{beamercolorbox}

	\sliderefs{Ref: \sliderefentry{1}{SANS Detection \& Response Survey (2024)}.}
\end{frame}

\begin{frame}{ML-based NIDS Workflow: Train $\rightarrow$ Validate $\rightarrow$ Test $\rightarrow$ Deploy}
	\small
	\begin{columns}[T,onlytextwidth]

		\column{0.55\linewidth}
		\begin{itemize}
			\setlength\itemsep{0.9em}

			\item<1-> The model is trained on a training set
			      \textbf{\(\mathcal{D}_{\text{train}}\)}.

			\item<2-> A separate validation split
			      \textbf{\(\mathcal{D}_{\text{val}}\)} to refine the model
			      \textbf{configuration} (\textbf{hyperparameters} and, in
			      anomaly detection, the \textbf{threshold}).

			\item<3-> A final test set,
			      \textbf{\(\mathcal{D}_{\text{test}}\)}, measures performance on unseen data.

			\item<4-> The selected model is deployed
			      to monitor live traffic.
		\end{itemize}

\begin{frame}{Focus of this thesis: robustness in ML-based NIDS}
	\small

	\vfill
	\begin{block}{Robustness in this thesis}
		\begin{itemize}
			\setlength\itemsep{0.6em}
			\item One fixed benchmark split is not enough.
			\item Robustness in operation: effectiveness under netwrok condition changes.
			\item Robustness in assessment: evaluation that truly stresses the detector.
		\end{itemize}

		\vspace{0.35em}
		\centering
		\textbf{$\Rightarrow$ Robustness = adaptation + rigorous evaluation.}
	\end{block}
	\vfill

\end{frame}
