\begin{frame}{From signatures to ML-based NIDS}
	\small
	\begin{columns}[T,onlytextwidth]
		\column{0.48\textwidth}
		\begin{block}{Rule-based NIDS}
			\begin{itemize}
				\setlength\itemsep{0.6em}
				\item Tools: \textbf{Snort}, \textbf{Suricata}, \textbf{Zeek}
				\item Detect only known attack patterns or protocol violations, leaving unknown ones undetected.
				\item Limited when payload content is encrypted
			\end{itemize}
		\end{block}

		\column{0.48\textwidth}
		\begin{block}{ML-based NIDS}
			\begin{itemize}
				\setlength\itemsep{0.6em}
				\item Example: \textbf{Random Forest (RF) on flows}
				\item Learn traffic patterns from historical data to generalize beyond exact rules.
				\item Can still detect suspicious behavior in encrypted traffic
			\end{itemize}
		\end{block}
	\end{columns}

	\vspace{0.25em}
	\begin{beamercolorbox}[wd=\linewidth,sep=0.9ex,rounded=true]{block body}
		\centering
		\textbf{Operational reality: mostly hybrid.} For example, a SOC may use \textbf{Suricata/Zeek} for known alerts and \textbf{ML on flows} to flag suspicious HTTPS command-and-control behavior.
	\end{beamercolorbox}

	\sliderefs{Refs: Snort [Roesch, 1999]; Zeek [Paxson, 1999]; Suricata (OISF); ML for NIDS [Sommer \& Paxson, 2010]; encrypted traffic [Anderson \& McGrew, 2016]; SANS Detection \& Response Survey (2024).}
\end{frame}

\begin{frame}{ML-based NIDS Workflow: Train $\rightarrow$ Validate $\rightarrow$ Test $\rightarrow$ Deploy}
	\small
	\begin{columns}[T,onlytextwidth]

		\column{0.55\linewidth}
		\begin{itemize}
			\setlength\itemsep{0.9em}

			\item<1-> The model is trained on a labelled \textbf{training} set
			      \(\mathcal{D}_{\text{train}}\).

			\item<2-> A separate \textbf{validation} split
			      \(\mathcal{D}_{\text{val}}\) is used to refine the model
			      \textbf{configuration} (\textbf{hyperparameters} and, in anomaly detection, the \textbf{threshold}).

			\item<3-> A final \textbf{test} set,
			      \(\mathcal{D}_{\text{test}}\), measures performance on previously unseen data.

			\item<4-> The selected model is then \textbf{deployed}
			      to monitor live traffic.
		\end{itemize}

		\only<2>{%
			\vspace{-7em}
			\setbeamercolor{focusbox}{use={structure,normal text},
				fg=structure.fg,
				bg=structure.fg!12!normal text.bg}
			\begin{beamercolorbox}[wd=\linewidth,sep=1ex,rounded=true]{focusbox}
				\small
				\textbf{Hyperparameters:} settings chosen before training that
				influence model behavior.\\
				\textbf{Example}: In RF, too few trees can give unstable
				predictions, while too many can increase computation and overfit
				the training data.
			\end{beamercolorbox}
		}

\begin{frame}{How ML has been used to improve NIDS}
	\fontsize{8.3}{9.2}\selectfont
	\begin{columns}[T,onlytextwidth]
		\column{0.56\linewidth}
		\begin{itemize}
			\item[\aspectbadge{violet!80!black}{F}] Richer traffic representations $\longrightarrow$ Flow-MAE, NetVigil\\[0.2em]
				      \hspace*{1.8em}{\fontsize{7.2}{7.9}\selectfont e.g., packet, flow, sequence, graph}
			\item[\aspectbadge{blue!75!black}{L}] Better use of limited labels $\longrightarrow$ Active-SVDD, INSOMNIA\\[0.2em]
				      \hspace*{1.8em}{\fontsize{7.2}{7.9}\selectfont e.g., active learning, semi-supervised learning, and pseudo-labeling}
			\item[\aspectbadge{orange!85!black}{P}] Better detection performance $\longrightarrow$ BotMiner, Kitsune\\[0.2em]
				      \hspace*{1.8em}{\fontsize{7.2}{7.9}\selectfont e.g., DDoS, botnet, web attacks}
			\item[\aspectbadge{green!60!black}{D}] More practical deployment $\longrightarrow$ AOC-IDS, ENIDrift\\[0.2em]
				      \hspace*{1.8em}{\fontsize{7.2}{7.9}\selectfont e.g., lightweight, online, real-time detection}
			\item[\aspectbadge{red!75!black}{R}] Better robustness in practice $\longrightarrow$ CADE, OWAD\\[0.2em]
				      \hspace*{1.8em}{\fontsize{7.2}{7.9}\selectfont e.g., adaptability to evolving traffic, resilience to adversarial attacks, and more challenging evaluation}
		\end{itemize}

	\end{columns}

\begin{frame}{Focus of this thesis: robustness in ML-based NIDS}
	\small

	\vfill
	\begin{block}{Robustness in this thesis}
		\begin{itemize}
			\setlength\itemsep{0.6em}
			\item Good performance on one fixed benchmark split is not sufficient to claim robustness.
			\item Before deployment, robustness depends on evaluation conditions that reveal detector weaknesses.
			\item After deployment, robustness means remaining effective when traffic and attack patterns evolve.
		\end{itemize}

		\vspace{0.35em}
		\centering
		\textbf{$\Rightarrow$ ML-based NIDS robustness is studied both before deployment, through the evaluation process, and after deployment, through adaptability to evolving traffic.}
	\end{block}
	\vfill

\end{frame}
