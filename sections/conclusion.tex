\begin{frame}{Domain Perspectives}
	\small

	{\normalsize\textbf{General takeaway}}\\[-0.05em]
	\begin{itemize}
		\setlength\itemsep{0.35em}
		\item Robust ML-based NIDS require both stronger pre-deployment evaluation and adaptive post-deployment configuration.
	\end{itemize}

	\vspace{0.2em}
	{\normalsize\textbf{Perspectives}}\\[-0.05em]
	\begin{itemize}
		\setlength\itemsep{0.35em}
		\item Move from one-shot benchmarking to continuous evaluation and adaptation loops.
		\item Neuro-symbolic NIDS are a promising direction for robust and explainable detection.
		\item Interpretable rules can encode security invariants, support stress testing, and bound safer reconfiguration.
		\item GenAI can help generate realistic, targeted test cases and reconfiguration candidates.
	\end{itemize}
\end{frame}
