\section{Part II: Auto-configuration}

\begin{frame}{Why ML-based NIDS still need explicit adaptation}
	\small

	\begin{columns}[T,onlytextwidth]
		\column{0.42\linewidth}
		\begin{itemize}
			\item ML-based NIDS are trained on historical traffic and deployed as static detectors.
			\item This assumes training and deployment traffic remain
			statistically close.
			\item Benign traffic evolves, attackers adapt \(\rightarrow\) the
			      network context changes.
		\end{itemize}

		\column{0.54\linewidth}
		\centering
		\includegraphics[width=\linewidth,height=0.46\textheight,keepaspectratio]{figures/auto_configuration/qiao_concept_drift.pdf}

		\par\vspace{0.1em}
		\figsource{Qiao et al., IEEE TII 2022, Table V/VI and Section IV-C}
	\end{columns}

	\vspace{0.6em}
	\begin{beamercolorbox}[wd=\linewidth,sep=0.9ex,rounded=true,center]{block body}
		\small
		\textbf{$\Rightarrow$ Detection performance can degrade $\Rightarrow$ explicit adaptation mechanisms are needed}\\[0.35em]
		\textbf{Adaptation} = post-deployment update of some part of the detection pipeline to maintain performance under network context change.\\[-0.02em]
		{\footnotesize e.g., the input data, the model, the hyperparameter configuration, or the threshold}
	\end{beamercolorbox}
\end{frame}

\begin{frame}{Is hyperparameter configuration important?}
	\small

	\begin{columns}[T,onlytextwidth]
		\column{0.42\linewidth}
		\begin{itemize}
			\footnotesize
			\setlength\itemsep{0.45em}
			\item We test whether a configuration optimized for one day can be reused on another day.
			\item Each boxplot summarizes the \textbf{performance decrease} when transferring the optimal configuration from other days.
			\item Some days show a limited median loss, but others reach about \textbf{20--40\%} degradation.
		\end{itemize}

		\column{0.56\linewidth}
		\centering
		\includegraphics[width=\linewidth]{figures/auto_configuration/noms2026_daily_transfer_configs_plot_rgba.png}

		\par\vspace{0.1em}
		\figsource{Anser et al., IEEE/IFIP NOMS 2026, Figure 5}
	\end{columns}

	\vspace{0.15em}
	\centering
	\footnotesize\textbf{$\Rightarrow$ Therefore, configuration transfer can help in some contexts, but it is not reliable as a general strategy.}

\end{frame}

\begin{frame}{How is configuration handled today?}
	\small

	\begin{block}{Conventional offline HPO}
		\begin{itemize}
			\setlength\itemsep{0.45em}
			\item Grid search, random search, Bayesian optimization, and evolutionary search.
			\item Repeated train-and-validate evaluations over candidate configurations.
		\end{itemize}
	\end{block}

	\vspace{0.35em}
	\begin{block}{Why not for ML-based NIDS adaptation?}
		\footnotesize
		These methods are static, offline, computationally expensive, and time-consuming procedures
		that must be re-run from scratch when the network context changes.
	\end{block}

	\vspace{0.25em}
	\begin{block}{A NIDS-specific example}
		\footnotesize
		In Anser et al. (NOMS 2023), Bayesian optimization of a binary Random Forest NIDS used
		\textbf{300 iterations} for a single daily network context, which corresponds to about
		\textbf{50 minutes} when one HPO iteration takes up to \textbf{10 seconds}.
	\end{block}

	\vspace{0.18em}
	\centering
	{\fontsize{5.8}{6.5}\selectfont Refs: Bergstra \& Bengio, JMLR 2012; Snoek et al., NeurIPS 2012; Anser et al., NOMS 2023.}
\end{frame}

\begin{frame}{Research question (RQ1)}
	\small

	\begin{block}{Research question}
		How can we find a configuration of a ML-based NIDS in a timely manner while maintaining an acceptable level of detection performance?
	\end{block}

	\vspace{0.35em}
	\begin{block}{From our analysis of hyperparameter impact}
		\begin{itemize}
			\item A NIDS configuration method has to be applied at each network context change.
			\item A new network context is defined by a new attack in the network or by a topology change.
		\end{itemize}
		\vspace{0.15em}
		{\scriptsize\emph{Source:} Anser et al., \emph{An Empirical Study on Configuration Robustness of Unsupervised IDS}, IEEE/IFIP NOMS 2026.}
	\end{block}

	\vspace{0.2em}
	\centering
	\small\textbf{$\Rightarrow$ Auto-configuration via meta-learning}
\end{frame}

\begin{frame}{Auto-configuration: conclusion}
	\small

	\begin{itemize}
		\setlength\itemsep{0.55em}
		\item Re-running HPO after each context change is too expensive for adaptive ML-based NIDS.
		\item Meta-learning preserves strong detection performance while staying close to HPO in both settings.
		\item The operational benefit is time: visible in supervised NIDS and decisive in attack-free AD.
		\item For AD NIDS, auto-configuration must remain context-aware, since both model HPs and threshold must be adapted.
	\end{itemize}

	\vspace{0.15em}
	\centering
	\footnotesize\textbf{$\Rightarrow$ Meta-learning makes NIDS reconfiguration fast enough for practice without sacrificing too much detection performance.}
\end{frame}

\begin{frame}{General conclusion}
	\small

	\begin{itemize}
		\setlength\itemsep{0.6em}
		\item Robust ML-based NIDS should not be judged only by their accuracy on a static benchmark.
		\item This thesis addressed robustness along two complementary axes: \textbf{configuration robustness} and \textbf{evaluation robustness}.
		\item Meta-learning enables fast, context-aware reconfiguration, including attack-free AD where both model HPs and threshold must be adapted.
		\item TATA strengthens pre-deployment evaluation by assessing test-set difficulty and generating targeted, protocol-faithful traffic to expose hidden weaknesses.
	\end{itemize}

	\vspace{0.15em}
	\centering
	\footnotesize\textbf{$\Rightarrow$ Robust ML-based NIDS require both adaptive configuration and stronger pre-deployment evaluation.}
\end{frame}
