\section{Part I: Adaptation}

\begin{frame}{Why ML-based NIDS still need explicit adaptation}
	\small

	\begin{columns}[T,onlytextwidth]
		\column{0.42\linewidth}
		\begin{itemize}
			\item Historical traffic for training; static detector at deployment.
			\item Assumes training and deployment traffic are statistically identical.
			\item Benign traffic evolves, attackers adapt \(\rightarrow\) the
			      network context changes.
		\end{itemize}

		\column{0.54\linewidth}
		\centering
		\includegraphics[width=\linewidth,height=0.46\textheight,keepaspectratio]{figures/auto_configuration/qiao_concept_drift.pdf}

		\par\vspace{0.1em}
		\figsource{Qiao et al., IEEE TII 2022, Table V/VI and Section IV-C}
	\end{columns}

	\vspace{0.6em}
	\begin{beamercolorbox}[wd=\linewidth,sep=0.9ex,rounded=true,center]{block body}
		\small
		\textbf{$\Rightarrow$ Detection performance can degrade $\Rightarrow$ explicit adaptation mechanisms are needed}\\[0.35em]
		\textbf{Adaptation} = post-deployment update of (data $|$ model $|$ config $|$ threshold) to maintain performance as the network context changes.
	\end{beamercolorbox}
\end{frame}

\begin{frame}{Is hyperparameter configuration important?}
	\small

	\begin{columns}[T,onlytextwidth]
		\column{0.42\linewidth}
		\begin{itemize}
			\footnotesize
			\setlength\itemsep{0.45em}
			\item Can a day-$t$ optimal config be reused on day $t'$?

		\end{itemize}

\begin{frame}{How is configuration handled today?}
	\small

	{\normalsize\textbf{Methods}}\\[-0.05em]
	\begin{itemize}
		\footnotesize
		\setlength\itemsep{0.15em}
		\item Grid search, random search\sliderefmark{1}, Bayesian optimization (BO)\sliderefmark{2}, evolutionary search.
		\item Repeated train-and-validate over candidate configurations.
	\end{itemize}

	\vspace{0.1em}
	{\normalsize\textbf{Main drawback}}\\[-0.05em]
	\begin{itemize}
		\footnotesize
		\setlength\itemsep{0.15em}
		\item Offline and expensive.
		\item Restarted from scratch after each context change.
	\end{itemize}

	\vspace{0.1em}
	{\normalsize\textbf{Real NIDS example\sliderefmark{3}}}\\[0.15em]
	\begin{itemize}
		\footnotesize
		\setlength\itemsep{0.1em}
		\item Bayesian optimization + binary Random Forest NIDS.
	\end{itemize}

	\vspace{0.18em}
	\begin{columns}[T,onlytextwidth]
		\column{0.31\linewidth}
		\begin{beamercolorbox}[rounded=true,shadow=true,wd=\linewidth,sep=0.8ex,center]{defbox}
			\Large\textbf{300}\\[-0.15em]
			\scriptsize iterations
		\end{beamercolorbox}

		\column{0.31\linewidth}
		\begin{beamercolorbox}[rounded=true,shadow=true,wd=\linewidth,sep=0.8ex,center]{defbox}
			\Large\textbf{10 s}\\[-0.15em]
			\scriptsize per iteration
		\end{beamercolorbox}

		\column{0.34\linewidth}
		\begin{beamercolorbox}[rounded=true,shadow=true,wd=\linewidth,sep=0.8ex,center]{defbox}
			\Large\textbf{$\approx$ 50 min}\\[-0.15em]
			\scriptsize per daily context
		\end{beamercolorbox}
	\end{columns}

	\vspace{0.15em}
	\centering
	\footnotesize\textbf{$\Longrightarrow$ Too slow for repeated post-deployment reconfiguration.}

	\vspace{0.14em}
	{\fontsize{5.8}{6.5}\selectfont Refs: \sliderefentry{1}{Bergstra \& Bengio, JMLR 2012}; \sliderefentry{2}{Snoek et al., NeurIPS 2012}; \sliderefentry{3}{Anser et al., NOMS 2023}.}
\end{frame}

\begin{frame}{Research question (RQ1)}
	\small

	\begin{block}{Research question}
		How can we find a configuration of a ML-based NIDS in a timely manner while maintaining an acceptable level of detection performance?
	\end{block}

	\vspace{0.35em}
	\begin{block}{From our analysis of hyperparameter impact}
		\begin{itemize}
			\item A NIDS configuration method has to be applied at each network context change.
			\item A new network context is defined by a new attack in the network or by a topology change.
		\end{itemize}
		\vspace{0.15em}
		{\scriptsize\emph{Source:} Anser et al., \emph{An Empirical Study on Configuration Robustness of Unsupervised IDS}, IEEE/IFIP NOMS 2026.}
	\end{block}

	\vspace{0.2em}
	\centering
	\small\textbf{$\Rightarrow$ Auto-configuration via meta-learning}
\end{frame}
